## Send Your First Payment

<CodeGroup>

```javascript Node.js
import Payman from "paymanai";

const payman = new Payman({
    xPaymanAPISecret: "<PAYMAN_API_KEY>",
    environment: "sandbox"
});

// Create a payee
const payee = await payman.payments.createPayee({
    type: "US_ACH",
    name: "John Doe",
    accountHolderName: "John Doe",
    accountHolderType: "individual",
    accountNumber: "12345678",
    routingNumber: "021000021",
    accountType: "checking",
    contactDetails: { 
        email: "john@example.com" 
    }
});

// Send payment
const payment = await payman.payments.sendPayment({
    amountDecimal: 100.00,
    paymentDestinationId: payee.id,
    memo: "Test payment"
});

console.log('Payment created:', payment);
```

```python Python
from paymanai import Paymanai

payman = Paymanai(
    x_payman_api_secret="<PAYMAN_API_KEY>",
    environment="sandbox"
)

// Create a payee
payee = payman.payments.create_payee(
    type="US_ACH",
    name="John Doe",
    account_holder_name="John Doe",
    account_holder_type="individual",
    account_number="12345678",
    routing_number="021000021",
    account_type="checking",
    contact_details={"email": "john@example.com"}
)

// Send payment
payment = payman.payments.send_payment(
    amount_decimal=100.00,
    payment_destination_id=payee["id"],
    memo="Test payment"
)

print('Payment created:', payment)
```
